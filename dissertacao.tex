% My environments
%Automatically numerate and label table rows.
%WARNING - must be used one counter per table
\newcounter{rowno}  %counter for the numbering
\newenvironment{row_labeled_longtable}[3] %two mandatory arguments - table name and disposition, as in the longtable environment
{
\begin{flushleft}   %align table to the left, because it is going to be wide
    %refstepcounter steps the counter and uses it for the following reference
    %start a longtable environment, supress left and right borders, and on the first column put "C(counter)" and the label (wich depends on the counter). Then format the columns according to the argument passed.
    %Rows can only be appended to the table to maintain compatibility
    \begin{longtable}{ @{} >{\refstepcounter{rowno}#3\therowno\label{#1:#3\therowno}} #2 @{} }
}
{
\end{longtable}
\end{flushleft}
}

\newcounter{rownos}  %counter for the numbering
\newenvironment{row_labeled_tabular}[3] %two mandatory arguments - table name and disposition, as in the longtable environment
{
%\begin{flushleft}   %align table to the left, because it is going to be wide
    %refstepcounter steps the counter and uses it for the following reference
    %start a longtable environment, supress left and right borders, and on the first column put "C(counter)" and the label (wich depends on the counter). Then format the columns according to the argument passed.
    %Rows can only be appended to the table to maintain compatibility
    \begin{tabular}{ @{} >{\refstepcounter{rownos}#3\therownos\label{#1:#3\therownos}} #2 @{} }
}
{
\end{tabular}
%\end{flushleft}
}
